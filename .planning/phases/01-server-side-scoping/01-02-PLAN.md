// Resolve active access.json: project-local overrides global when present.
// DISCORD_PROJECT_DIR is injected by Claude Code via .mcp.json env block (${CLAUDE_PROJECT_DIR}).
const PROJECT_DIR = process.env.DISCORD_PROJECT_DIR

function resolveAccessFile(): { path: string; scope: 'local' | 'global' } {
  if (PROJECT_DIR) {
    // Security: ensure resolved path is actually inside the stated project dir.
    const candidate = join(PROJECT_DIR, '.claude', 'channels', 'discord', 'access.json')
    if (!candidate.startsWith(PROJECT_DIR + '/') && candidate !== PROJECT_DIR) {
      // Path traversal attempt — fall back silently.
      process.stderr.write(`discord channel: DISCORD_PROJECT_DIR resolved outside project boundary, falling back to global\n`)
      return { path: ACCESS_FILE, scope: 'global' }
    }
    try {
      statSync(candidate)  // throws ENOENT if absent
      return { path: candidate, scope: 'local' }
    } catch (e) {
      if ((e as NodeJS.ErrnoException).code !== 'ENOENT') {
        process.stderr.write(`discord channel: unexpected error stat-ing local access.json: ${e}\n`)
      }
      // ENOENT is normal — fall through to global
    }
  } else {
    process.stderr.write(`discord channel: DISCORD_PROJECT_DIR not set — project scoping unavailable, using global config\n`)
  }
  return { path: ACCESS_FILE, scope: 'global' }
}

const { path: ACTIVE_ACCESS_FILE, scope: ACTIVE_SCOPE } = resolveAccessFile()
process.stderr.write(`discord channel: using ${ACTIVE_SCOPE} config ${ACTIVE_ACCESS_FILE}\n`)
